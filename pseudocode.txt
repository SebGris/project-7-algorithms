Algorithme MaximiserProfit
    Entrée :
        - ListeActions : une liste d'actions disponibles avec leurs coûts et profits estimés après deux ans
        - BudgetMax : le budget maximal par client (500 euros)
    Sortie :
        - MeilleurPortefeuille : une liste d'actions à acheter pour maximiser le profit

    Fonction CalculerProfit(portefeuille)
        profitTotal ← 0
        Pour chaque action dans portefeuille
            profitTotal ← profitTotal + action.profitEstimé
        Fin Pour
        Retourner profitTotal
    Fin Fonction

    Fonction EstValide(portefeuille)
        coûtTotal ← 0
        Pour chaque action dans portefeuille
            coûtTotal ← coûtTotal + action.coût
        Fin Pour
        Si coûtTotal ≤ BudgetMax alors
            Retourner Vrai
        Sinon
            Retourner Faux
        Fin Si
    Fin Fonction

    Procédure GénérerCombinations(ListeActions, index, combinaisonActuelle, toutesCombinations)
        Si index = longueur(ListeActions) alors
            Ajouter combinaisonActuelle à toutesCombinations
            Retourner
        Fin Si

        // Inclure l'action courante dans la combinaison
        GénérerCombinations(ListeActions, index + 1, combinaisonActuelle + [ListeActions[index]], toutesCombinations)

        // Exclure l'action courante de la combinaison
        GénérerCombinations(ListeActions, index + 1, combinaisonActuelle, toutesCombinations)
    Fin Procédure

    Fonction TrouverMeilleurPortefeuille(ListeActions, BudgetMax)
        MeilleurPortefeuille ← []
        MeilleurProfit ← 0
        toutesCombinations ← []

        // Générer toutes les combinaisons possibles d'actions
        GénérerCombinations(ListeActions, 0, [], toutesCombinations)

        Pour chaque combinaison dans toutesCombinations
            Si EstValide(combinaison) alors
                profit ← CalculerProfit(combinaison)
                Si profit > MeilleurProfit alors
                    MeilleurProfit ← profit
                    MeilleurPortefeuille ← combinaison
                Fin Si
            Fin Si
        Fin Pour

        Retourner MeilleurPortefeuille
    Fin Fonction

    Début
        ListeActions ← LireListeActions()  // Lire la liste des actions disponibles
        BudgetMax ← 500  // Budget maximal par client
        MeilleurPortefeuille ← TrouverMeilleurPortefeuille(ListeActions, BudgetMax)
        Afficher("Le portefeuille qui maximise le profit est : ", MeilleurPortefeuille)
    Fin
Fin Algorithme
